Держи пример сортировки на Python:

# Быстрая сортировка списка
numbers = [5, 2, 8, 1, 9, 3]

# Способ 1: встроенная сортировка
sorted_numbers = sorted(numbers)  # [1, 2, 3, 5, 8, 9]

# Способ 2: сортировка по убыванию
sorted_desc = sorted(numbers, reverse=True)  # [9, 8, 5, 3, 2, 1]

# Способ 3: сортировка пузырьком (для понимания)
def bubble_sort(arr):
    n = len(arr)
    for i in range(n-1):
        for j in range(n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

print(bubble_sort([5, 2, 8, 1, 9, 3]))  # [1, 2, 3, 5, 8, 9]
